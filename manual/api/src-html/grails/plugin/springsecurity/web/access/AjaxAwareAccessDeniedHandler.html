015package grails.plugin.springsecurity.web.access;
016
017import grails.plugin.springsecurity.SpringSecurityUtils;
018
019import java.io.IOException;
020
021import javax.servlet.ServletException;
022import javax.servlet.http.HttpServletRequest;
023import javax.servlet.http.HttpServletResponse;
024
025import org.springframework.beans.factory.InitializingBean;
026import org.springframework.security.access.AccessDeniedException;
027import org.springframework.security.authentication.AuthenticationTrustResolver;
028import org.springframework.security.core.Authentication;
029import org.springframework.security.core.context.SecurityContextHolder;
030import org.springframework.security.web.PortResolver;
031import org.springframework.security.web.WebAttributes;
032import org.springframework.security.web.access.AccessDeniedHandler;
033import org.springframework.security.web.savedrequest.RequestCache;
034import org.springframework.util.Assert;
035
036/**
037 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
038 */
039public class AjaxAwareAccessDeniedHandler implements AccessDeniedHandler, InitializingBean {
040
041        protected String errorPage;
042        protected String ajaxErrorPage;
043        protected PortResolver portResolver;
044        protected AuthenticationTrustResolver authenticationTrustResolver;
045        protected boolean useForward = true;
046        protected RequestCache requestCache;
047
048        /**
049         * {@inheritDoc}
050         * @see org.springframework.security.web.access.AccessDeniedHandler#handle(
051         *      javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse,
052         *      org.springframework.security.access.AccessDeniedException)
053         */
054        public void handle(final HttpServletRequest request, final HttpServletResponse response,
055                        final AccessDeniedException e) throws IOException, ServletException {
056
057                if (e != null && isLoggedIn() && authenticationTrustResolver.isRememberMe(getAuthentication())) {
058                        // user has a cookie but is getting bounced because of IS_AUTHENTICATED_FULLY,
059                        // so Spring Security won't save the original request
060                        requestCache.saveRequest(request, response);
061                }
062
063                if (response.isCommitted()) {
064                        return;
065                }
066
067                boolean ajaxError = ajaxErrorPage != null && SpringSecurityUtils.isAjax(request);
068                if (errorPage == null && !ajaxError) {
069                        response.sendError(HttpServletResponse.SC_FORBIDDEN, e.getMessage());
070                        return;
071                }
072
073                if (useForward && (errorPage != null || ajaxError)) {
074                        // Put exception into request scope (perhaps of use to a view)
075                        request.setAttribute(WebAttributes.ACCESS_DENIED_403, e);
076                        response.setStatus(HttpServletResponse.SC_FORBIDDEN);
077                        request.getRequestDispatcher(ajaxError ? ajaxErrorPage : errorPage).forward(request, response);
078                        return;
079                }
080
081                boolean includePort = true;
082                String scheme = request.getScheme();
083                String serverName = request.getServerName();
084                int serverPort = portResolver.getServerPort(request);
085                String contextPath = request.getContextPath();
086                boolean inHttp = "http".equals(scheme.toLowerCase());
087                boolean inHttps = "https".equals(scheme.toLowerCase());
088
089                if (inHttp && (serverPort == 80)) {
090                        includePort = false;
091                }
092                else if (inHttps && (serverPort == 443)) {
093                        includePort = false;
094                }
095
096                String redirectUrl = scheme + "://" + serverName + ((includePort) ? (":" + serverPort) : "") + contextPath;
097                if (ajaxError) {
098                        redirectUrl += ajaxErrorPage;
099                }
100                else if (errorPage != null) {
101                        redirectUrl += errorPage;
102                }
103                response.sendRedirect(response.encodeRedirectURL(redirectUrl));
104        }
105
106        protected Authentication getAuthentication() {
107                return SecurityContextHolder.getContext() == null ? null :
108                       SecurityContextHolder.getContext().getAuthentication();
109        }
110
111        protected boolean isLoggedIn() {
112                Authentication authentication = getAuthentication();
113                if (authentication == null) {
114                        return false;
115                }
116                return !authenticationTrustResolver.isAnonymous(authentication);
117        }
118
119        /**
120         * Dependency injection for the error page, e.g. '/login/denied'.
121         * @param page the page
122         */
123        public void setErrorPage(final String page) {
124                Assert.isTrue(page == null || page.startsWith("/"), "ErrorPage must begin with '/'");
125                errorPage = page;
126        }
127
128        /**
129         * Dependency injection for the Ajax error page, e.g. '/login/ajaxDenied'.
130         * @param page the page
131         */
132        public void setAjaxErrorPage(final String page) {
133                Assert.isTrue(page == null || page.startsWith("/"), "Ajax ErrorPage must begin with '/'");
134                ajaxErrorPage = page;
135        }
136
137        /**
138         * Dependency injection for the port resolver.
139         * @param resolver the resolver
140         */
141        public void setPortResolver(final PortResolver resolver) {
142                portResolver = resolver;
143        }
144
145        /**
146         * Dependency injection for the {@link AuthenticationTrustResolver}.
147         * @param resolver the resolver
148         */
149        public void setAuthenticationTrustResolver(final AuthenticationTrustResolver resolver) {
150                authenticationTrustResolver = resolver;
151        }
152
153        /**
154         * Dependency injection for whether to forward or redirect.
155         * @param forward if <code>true</code> forward to render the denied page, otherwise redirect
156         */
157        public void setUseForward(boolean forward) {
158                useForward = forward;
159        }
160
161        /**
162         * Dependency injection for the request cache.
163         * @param cache the cache
164         */
165        public void setRequestCache(RequestCache cache) {
166                requestCache = cache;
167        }
168
169        /**
170         * {@inheritDoc}
171         * @see org.springframework.beans.factory.InitializingBean#afterPropertiesSet()
172         */
173        public void afterPropertiesSet() {
174                Assert.notNull(portResolver, "portResolver is required");
175                Assert.notNull(authenticationTrustResolver, "authenticationTrustResolver is required");
176                Assert.notNull(requestCache, "requestCache is required");
177        }
178}
