015package grails.plugin.springsecurity.web.access.intercept;
016
017import grails.plugin.springsecurity.InterceptedUrl;
018import grails.plugin.springsecurity.ReflectionUtils;
019
020import java.util.List;
021import java.util.Map;
022
023/**
024 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
025 */
026public class InterceptUrlMapFilterInvocationDefinition extends AbstractFilterInvocationDefinition {
027
028        @Override
029        protected void initialize() {
030                if (!initialized) {
031                        reset();
032                }
033        }
034
035        @Override
036        protected boolean stopAtFirstMatch() {
037                return true;
038        }
039
040        @SuppressWarnings("unchecked")
041        @Override
042        public void reset() {
043                Object map = ReflectionUtils.getConfigProperty("interceptUrlMap");
044                if (!(map instanceof Map || map instanceof List)) {
045                        log.warn("interceptUrlMap config property isn't a Map or a List of Maps");
046                        return;
047                }
048
049                resetConfigs();
050
051                List<InterceptedUrl> data = map instanceof Map ? ReflectionUtils.splitMap((Map<String, Object>)map) :
052                                                                 ReflectionUtils.splitMap((List<Map<String, Object>>)map);
053                for (InterceptedUrl iu : data) {
054                        compileAndStoreMapping(iu);
055                }
056
057                initialized = true;
058        }
059}
