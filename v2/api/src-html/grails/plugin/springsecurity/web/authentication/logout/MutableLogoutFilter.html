015package grails.plugin.springsecurity.web.authentication.logout;
016
017import java.io.IOException;
018import java.util.List;
019
020import javax.servlet.FilterChain;
021import javax.servlet.ServletException;
022import javax.servlet.ServletRequest;
023import javax.servlet.ServletResponse;
024import javax.servlet.http.HttpServletRequest;
025import javax.servlet.http.HttpServletResponse;
026
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029import org.springframework.security.core.Authentication;
030import org.springframework.security.core.context.SecurityContextHolder;
031import org.springframework.security.web.authentication.logout.LogoutFilter;
032import org.springframework.security.web.authentication.logout.LogoutHandler;
033import org.springframework.security.web.authentication.logout.LogoutSuccessHandler;
034
035/**
036 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
037 */
038public class MutableLogoutFilter extends LogoutFilter {
039
040        protected final LogoutSuccessHandler logoutSuccessHandler;
041        protected final Logger log = LoggerFactory.getLogger(getClass());
042
043        protected List<LogoutHandler> handlers;
044
045        /**
046         * Constructor.
047         * @param successHandler the logout success handler
048         */
049        public MutableLogoutFilter(LogoutSuccessHandler successHandler) {
050                super(successHandler, new DummyLogoutHandler());
051                logoutSuccessHandler = successHandler;
052        }
053
054        @Override
055        public void doFilter(final ServletRequest req, final ServletResponse res, final FilterChain chain)
056                        throws IOException, ServletException {
057
058                HttpServletRequest request = (HttpServletRequest) req;
059                HttpServletResponse response = (HttpServletResponse) res;
060
061                if (!requiresLogout(request, response)) {
062                        chain.doFilter(request, response);
063                        return;
064                }
065
066                Authentication auth = SecurityContextHolder.getContext().getAuthentication();
067                log.debug("Logging out user '{}' and transferring to logout destination", auth);
068
069                for (LogoutHandler handler : handlers) {
070                        handler.logout(request, response, auth);
071                }
072
073                logoutSuccessHandler.onLogoutSuccess(request, response, auth);
074        }
075
076        /**
077         * Dependency injection for the logout handlers.
078         * @param l the handlers
079         */
080        public void setHandlers(final List<LogoutHandler> l) {
081                handlers = l;
082        }
083
084        /**
085         * Null logout handler that's used to provide a non-empty list of handlers to the base class.
086         * The real handlers will be after construction.
087         */
088        protected static class DummyLogoutHandler implements LogoutHandler {
089                public void logout(HttpServletRequest request, HttpServletResponse response, Authentication authentication) {
090                        // do nothing
091                }
092        }
093}
