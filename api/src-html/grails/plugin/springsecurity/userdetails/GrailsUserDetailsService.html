015package grails.plugin.springsecurity.userdetails;
016
017import org.springframework.dao.DataAccessException;
018import org.springframework.security.core.userdetails.UserDetails;
019import org.springframework.security.core.userdetails.UserDetailsService;
020import org.springframework.security.core.userdetails.UsernameNotFoundException;
021
022/**
023 * Extension of the standard interface that allows specifying whether or not to load roles
024 * from the database, e.g. for LDAP where role information is inferred from LDAP group membership.
025 *
026 * @author <a href='mailto:burt@burtbeckwith.com'>Burt Beckwith</a>
027 */
028public interface GrailsUserDetailsService extends UserDetailsService {
029
030        /**
031         * Locates the user based on the username.
032         *
033         * @param username the username identifying the user whose data is required.
034         * @param loadRoles whether to load roles at the same time as loading the user
035         *
036         * @return a fully populated user record (never <code>null</code>)
037         *
038         * @throws UsernameNotFoundException if the user could not be found
039         * @throws DataAccessException if user could not be found for a repository-specific reason
040         */
041        UserDetails loadUserByUsername(String username, boolean loadRoles) throws UsernameNotFoundException, DataAccessException;
042}
